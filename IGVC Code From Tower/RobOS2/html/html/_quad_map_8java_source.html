00584          * 
00585          * from: Vincenty direct formula - T Vincenty, "Direct and Inverse Solutions
00586          * of Geodesics on the Ellipsoid with application of nested equations",
00587          * Survey Review, vol XXII no 176, 1975
00588          * http://www.ngs.noaa.gov/PUBS_LIB/inverse.pdf

00590         public double[] destination(double lat1, double lon1, double brng,
00591                         double dist) {

00593                                                                                                                                                 // ellipsiod
00594                 double s = dist;
00595                 double alpha1 = Math.toRadians(brng);
00596                 double sinAlpha1 = Math.sin(alpha1);
00597                 double cosAlpha1 = Math.cos(alpha1);
00598 
00599                 double tanU1 = (1 - f) * Math.tan(Math.toRadians(lat1));
00600                 double cosU1 = 1 / Math.sqrt((1 + tanU1 * tanU1)), sinU1 = tanU1
00601                                 * cosU1;
00602                 double sigma1 = Math.atan2(tanU1, cosAlpha1);
00603                 double sinAlpha = cosU1 * sinAlpha1;
00604                 double cosSqAlpha = 1 - sinAlpha * sinAlpha;
00605                 double uSq = cosSqAlpha * (a * a - b * b) / (b * b);
00606                 double A = 1 + uSq / 16384
00607                                 * (4096 + uSq * (-768 + uSq * (320 - 175 * uSq)));
00608                 double B = uSq / 1024 * (256 + uSq * (-128 + uSq * (74 - 47 * uSq)));
00609                 double sinSigma = 0.0;
00610                 double cosSigma = 0.0;
00611                 double cos2SigmaM = 0.0;
00612                 double deltaSigma = 0.0;
00613 
00614                 double sigma = s / (b * A), sigmaP = 2 * Math.PI;
00615                 while (Math.abs(sigma - sigmaP) > 1e-12) {
00616                         cos2SigmaM = Math.cos(2 * sigma1 + sigma);
00617                         sinSigma = Math.sin(sigma);
00618                         cosSigma = Math.cos(sigma);
00619                         deltaSigma = B
00620                                         * sinSigma
00621                                         * (cos2SigmaM + B
00622                                                         / 4
00623                                                         * (cosSigma * (-1 + 2 * cos2SigmaM * cos2SigmaM) - B
00624                                                                         / 6
00625                                                                         * cos2SigmaM
00626                                                                         * (-3 + 4 * sinSigma * sinSigma)
00627                                                                         * (-3 + 4 * cos2SigmaM * cos2SigmaM)));
00628                         sigmaP = sigma;
00629                         sigma = s / (b * A) + deltaSigma;
00630                 }
00631 
00632                 double tmp = sinU1 * sinSigma - cosU1 * cosSigma * cosAlpha1;
00633                 double lat2 = Math.atan2(sinU1 * cosSigma + cosU1 * sinSigma
00634                                 * cosAlpha1,
00635                                 (1 - f) * Math.sqrt(sinAlpha * sinAlpha + tmp * tmp));
00636                 double lambda = Math.atan2(sinSigma * sinAlpha1, cosU1 * cosSigma
00637                                 - sinU1 * sinSigma * cosAlpha1);
00638                 double C = f / 16 * cosSqAlpha * (4 + f * (4 - 3 * cosSqAlpha));
00639                 double L = lambda
00640                                 - (1 - C)
00641                                 * f
00642                                 * sinAlpha
00643                                 * (sigma + C
00644                                                 * sinSigma
00645                                                 * (cos2SigmaM + C * cosSigma
00646                                                                 * (-1 + 2 * cos2SigmaM * cos2SigmaM)));
00647 
00648                 //@SuppressWarnings("unused")
00649                 //double revAz = Math.atan2(sinAlpha, -tmp); // final bearing
00650                 double[] Coords = new double[2];
00651                 Coords[0] = Math.toDegrees(lat2);
00652                 Coords[1] = Math.toDegrees(lon1 + L);
00653                 return Coords;
